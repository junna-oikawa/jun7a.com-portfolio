function calc(polygon) {
  let n = 0;
  centX = 0;
  centY = 0;
  polygon[polygon.length] = polygon[0];
  for(var j = 0; j < height; j++) {
    for(var i = 0; i < width; i++) {
      if (cn(polygon, i, j) == true) {
      centX += i;
      centY += j;
      n++;
      }
    }
  }
  centX /= n;
  centY /= n;
  drawCross(centX, centY);
}

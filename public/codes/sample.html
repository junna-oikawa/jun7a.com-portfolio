let maxDistance = 0;
let useEyelet;
// eyeletIdには、クリックした要素に付いている全てのハトメのidが格納されている
for (let i = 0; i < eyeletId.length; i++) {
  // 画面上から対象idのハトメを取得する
  let eyelet = stage.findOne('#' + eyeletId[i]);
  let thisEyeletPos = eyelet.getAbsolutePosition();
  let distance =
    Math.sqrt((thisEyeletPos.x - mousePos.x) ** 2 + (thisEyeletPos.y - mousePos.y) ** 2);
  if (distance > maxDistance) {
    maxDistance = distance;
    axis = thisEyeletPos;
    useEyelet = eyelet;
  }
}
